```yaml
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: arango-pvc
  namespace: arango-db
spec:
  storageClassName: managed-standard-retain-csi
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 10Gi
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: arango-db
  namespace: arango-db
spec:
  replicas: 1
  selector:
    matchLabels:
      app: arango-db
  template:
    metadata:
      labels:
        app: arango-db
    spec:
      containers:
      - name: arango
        image: arangodb:latest
        ports:
        - containerPort: 8529
        env:
        - name: ARANGO_ROOT_PASSWORD
          value: "rootpassword"
        volumeMounts:
        - name: arango-data
          mountPath: /var/lib/arangodb3
        - name: arango-apps-data
          mountPath: /var/lib/arangodb3-apps
      volumes:
      - name: arango-data
        persistentVolumeClaim:
          claimName: arango-pvc
      - name: arango-apps-data
        emptyDir: {} 
---
apiVersion: v1
kind: Service
metadata:
  name: arangodb-service
  namespace: arango-db
spec:
  type: ClusterIP
  ports:
    - port: 80
      targetPort: 8529
  selector:
    app: arango-db
```
